       \c@mm=\c@section%

       \advance\c@p by 1 %
       \advance\c@mm by -2 %
       \advance\c@pp by 2 %
       %
       \ifnum \c@section=1
                    \ifnum\c@section=##1%
               \setbox\beamer@tempbox=\hbox{%
              \hyperlink{Navigation##3}{\hbox to #1{%
             {\hskip0.3cm\usebeamertemplate{section in head/foot}\hskip0.3cm}}}}%

            \fi%%
